773 timeval NTPClient::calculateOffset (NTPPacket_t* ntpPacket) {
774  double t1, t2, t3, t4;
775  timeval tv_offset;
776  //timeval currenttime;
777 
778  t1 = ntpPacket->origin.tv_sec + ntpPacket->origin.tv_usec / 1000000.0;
779  t2 = ntpPacket->receive.tv_sec + ntpPacket->receive.tv_usec / 1000000.0;
780  t3 = ntpPacket->transmit.tv_sec + ntpPacket->transmit.tv_usec / 1000000.0;
781  t4 = ntpPacket->destination.tv_sec + ntpPacket->destination.tv_usec / 1000000.0;
782  offset = ((t2 - t1) / 2.0 + (t3 - t4) / 2.0); // in seconds
783  delay = (t4 - t1) - (t3 - t2); // in seconds
784 
785  DEBUGLOG ("T1: %f T2: %f T3: %f T4: %f", t1, t2, t3, t4);
786  DEBUGLOG ("T1: %s", getTimeDateString (ntpPacket->origin));
787  //DEBUGLOG ("T1: %016X %016X", ntpPacket->origin.tv_sec, ntpPacket->origin.tv_usec);
788  DEBUGLOG ("T2: %s", getTimeDateString (ntpPacket->receive));
789  //DEBUGLOG ("T2: %016X %016X", ntpPacket->receive.tv_sec, ntpPacket->receive.tv_usec);
790  DEBUGLOG ("T3: %s", getTimeDateString (ntpPacket->transmit));
791  //DEBUGLOG ("T3: %016X %016X", ntpPacket->transmit.tv_sec, ntpPacket->transmit.tv_usec);
792  DEBUGLOG ("T4: %s", getTimeDateString (ntpPacket->destination));
793  //DEBUGLOG ("T4: %016X %016X", ntpPacket->destination.tv_sec, ntpPacket->destination.tv_usec);
794  DEBUGLOG ("Offset: %f, Delay: %f", offset, delay);
795 
796  //DEBUGLOG ("Current time: %d.%d", currenttime.tv_sec, currenttime.tv_usec);
797  //DEBUGLOG ("Current time: %s", ctime (&(currenttime.tv_sec)));
798 
799  tv_offset.tv_sec = (time_t)offset;
800  tv_offset.tv_usec = (offset - (double)tv_offset.tv_sec) * 1000000.0;
801 
802  DEBUGLOG ("Calculated offset %f sec. Delay %f ms", offset, delay * 1000);
803

890  break;

905  e.info.port,
906  e.info.offset * 1000);
907  break;

919  break;

921  snprintf (result, resultMaxSize, "%d: Unknown error", e.event);
